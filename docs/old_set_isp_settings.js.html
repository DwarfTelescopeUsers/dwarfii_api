/**
 * 3.3.1 brightness
 * @param {number} camera
 * @param {number} value
 * @returns {Object}
 */
export function setBrightness(camera = telephotoCamera, value) {
  // telephoto values: 0-255 default：128
  // wideangle values: -64-64 default：0
  const options = {
    interface: setBrightnessValueCmd,
    camId: camera,
    value,
  };
  return options;
}
